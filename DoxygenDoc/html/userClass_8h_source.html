477  {
478  hv[i][j]=-1;

481 
482  for(int i=0;i<anchor;i++)
483  {
484  unordered_set<int> visited;
485  queue<int> nq;
486  nq.push(i);
487  while(!nq.empty())
488  {
489  int cur=nq.front();
490  nq.pop();
491  visited.insert(cur);

493  if(!neighborsList[cur].empty()){
494  for(int neighbor : neighborsList[cur])
495  {
496  if(visited.find(neighbor)==visited.end())
497  {
498  nq.push(neighbor);
499  visited.insert(neighbor);
500  }
501  if(hv[neighbor][i]==-1)hv[neighbor][i]=curHop+1;
502  else

504  hv[neighbor][i]=hv[neighbor][i]<curHop+1?hv[neighbor][i]:curHop+1;
