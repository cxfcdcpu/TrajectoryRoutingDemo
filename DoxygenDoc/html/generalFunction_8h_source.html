50 void updateTAS(twoCycleTrial ct, unordered_set<string> & TAS)
51 {

54  {
55  stringstream ss(st);
56  float i,j;
57  ss>>i;
58  ss>>j;
59  //<<i<<endl;
60  float di1=ct.c1X-i;
61  float dj1=ct.c1Y-j;
62  float di2=ct.c2X-i;
63  float dj2=ct.c2Y-j;
64  float rr1 = ct.h1*ct.d*ct.h1*ct.d;
65  float rr2 = ct.h2*ct.d*ct.h2*ct.d;
66  float rr3 = (ct.h1-1)*ct.d*(ct.h1-1)*ct.d;
67 
68  if (di1*di1+dj1*dj1<=rr1 && di1*di1+dj1*dj1>rr3 && di2*di2+dj2*dj2<=rr2)
69  {
70  //cout<<"erasing cycle tas"<<endl;
71  temp.push_back(st);
72  }
73  }

147 int lDis(Point p0, Point p1, Point p2)
148 {
149  if(p2.m_X!=p1.m_X && p2.m_Y != p1.m_Y)
150  return ((p2.m_X-p1.m_X)*(p1.m_Y-p0.m_Y)-(p1.m_X-p0.m_X)*(p2.m_Y-p1.m_Y))*((p2.m_X-p1.m_X)*(p1.m_Y-p0.m_Y)-(p1.m_X-p0.m_X)*(p2.m_Y-p1.m_Y))/((p2.m_X-p1.m_X)*(p2.m_X-p1.m_X)+(p2.m_Y-p1.m_Y)*(p2.m_Y-p1.m_Y));
151  else
152  return sqrt(pDis(p0,p1));
