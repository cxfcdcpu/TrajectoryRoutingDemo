252  float d=sqrt(pDis(p1,p2));

257

260  else if(r1>r2+d)A1=pi*r2*r2;
261 
262  if(A1==0)
263  {
264  if(d > 0 && r2 > 0 && r3 >0)
265  A1=r2*r2*acos((d*d+r2*r2-r1*r1) / (2*d*r2))+r1*r1*acos((d*d+r1*r1-r2*r2) / (2*d*r1))-0.5*sqrt((-d+r2+r1)*(d+r2-r1)*(d-r2+r1)*(d+r1+r2));
266  else
267  A1=10000;
268  }
269 
270  if (r3==0)return A1;
271  float A2=0;
272  if(d>=r3+r2)return A1;
273  else if(r2>=r3+d)return A1-pi*r3*r3;

276  A2=r2*r2*acos((d*d+r2*r2-r3*r3) / (2*d*r2))+r3*r3*acos((d*d+r3*r3-r2*r2) / (2*d*r3))-0.5*sqrt((-d+r2+r3)*(d+r2-r3)*(d-r2+r3)*(d+r3+r2));

279  return A1-A2;
